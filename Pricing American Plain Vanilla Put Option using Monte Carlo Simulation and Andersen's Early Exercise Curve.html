# early exercise if  S < K  and (T-t)/T < c0*((K-S)/K)^2
def myval(c):
    val = np.zeros(M)
    for i in range(1,N+1):
        con1 = S[:,i] < K
        con2 = (T-i*h)/T <  c[0]*((K-S[:,i])/K)**2 
        val =  val + con1*con2*(val==0)*(np.exp(-i*h*r) * (K-S[:,i]))
    dd = np.mean(val)
    err = np.std(val)/np.sqrt(M)
    print(c,dd,err)    
    return(-dd)
